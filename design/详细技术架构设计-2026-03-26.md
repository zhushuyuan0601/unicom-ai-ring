#### 核心接口定义

```go
// Go 语言示例

package receptionist

// AI接线员服务接口
type ReceptionistService interface {
    // 配置管理
    CreateConfig(ctx context.Context, userID string, config *ReceptionistConfig) error
    GetConfig(ctx context.Context, userID string) (*ReceptionistConfig, error)
    UpdateConfig(ctx context.Context, userID string, config *ReceptionistConfig) error
    
    // 来电处理
    HandleIncomingCall(ctx context.Context, call *IncomingCall) (*CallSession, error)
    ProcessAudio(ctx context.Context, sessionID string, audio []byte) (*ProcessResult, error)
    EndCall(ctx context.Context, sessionID string) (*CallSummary, error)
    
    // 通话记录
    GetCallHistory(ctx context.Context, userID string, opts *ListOptions) ([]*CallRecord, error)
    GetCallDetail(ctx context.Context, callID string) (*CallRecord, error)
    
    // 知识库
    AddKnowledge(ctx context.Context, userID string, knowledge *Knowledge) error
    SearchKnowledge(ctx context.Context, userID string, query string) ([]*Knowledge, error)
}

// 接线员配置
type ReceptionistConfig struct {
    ID            string                 `json:"id"`
    UserID        string                 `json:"userId"`
    Enabled       bool                   `json:"enabled"`
    Mode          string                 `json:"mode"`          // always/busy/schedule
    Schedule      *ScheduleConfig        `json:"schedule"`
    Greeting      *GreetingConfig        `json:"greeting"`
    SmartReply    *SmartReplyConfig      `json:"smartReply"`
    Voicemail     *VoicemailConfig       `json:"voicemail"`
    Advanced      *AdvancedConfig        `json:"advanced"`
    CreatedAt     time.Time              `json:"createdAt"`
    UpdatedAt     time.Time              `json:"updatedAt"`
}

// 问候语配置
type GreetingConfig struct {
    Type       string `json:"type"`       // default/custom
    CustomText string `json:"customText"`
    VoiceID    string `json:"voiceId"`
}

// 智能应答配置
type SmartReplyConfig struct {
    Enabled      bool     `json:"enabled"`
    Personality  string   `json:"personality"`  // friendly/professional/casual
    KnowledgeIDs []string `json:"knowledgeIds"`
}

// 来电信息
type IncomingCall struct {
    CallID       string    `json:"callId"`
    CallerNumber string    `json:"callerNumber"`
    CalleeNumber string    `json:"calleeNumber"`
    Timestamp    time.Time `json:"timestamp"`
}

// 通话会话
type CallSession struct {
    ID           string          `json:"id"`
    CallID       string          `json:"callId"`
    UserID       string          `json:"userId"`
    Status       string          `json:"status"`     // active/ended
    StartTime    time.Time       `json:"startTime"`
    Context      *DialogContext  `json:"context"`
    Transcripts  []Transcript    `json:"transcripts"`
}

// 对话上下文
```

```go
// AI速记服务接口
type ShorthandService interface {
    // 实时转写
    StartTranscription(ctx context.Context, callID string) (*TranscriptionSession, error)
    ProcessAudioChunk(ctx context.Context, sessionID string, chunk []byte) error
    EndTranscription(ctx context.Context, sessionID string) (*CallRecord, error)
    
    // 实时推送
    Subscribe(sessionID string) (<-chan TranscriptionUpdate, error)
    Unsubscribe(sessionID string, ch <-chan TranscriptionUpdate)
    
    // 后处理
    GenerateSummary(ctx context.Context, transcription string) (*Summary, error)
    ExtractTodos(ctx context.Context, transcription string) ([]*TodoItem, error)
    ExtractKeyInfo(ctx context.Context, transcription string) (*KeyInfo, error)
    
    // 查询
    GetRecords(ctx context.Context, userID string, opts *ListOptions) ([]*CallRecord, error)
    GetRecord(ctx context.Context, recordID string) (*CallRecord, error)
    SearchRecords(ctx context.Context, query string, opts *SearchOptions) ([]*CallRecord, error)
}
```
